//Print example

/*
Функция Print так же присутствует как глобальная. Поэтому для различия имён подставлено
нижнее подчёркивание.
При существовании двух вариантов функции наибольшим приоритетом 
обладает пользовательская (определённая в скрипте, а не в главной программе)
*/


def _print()
	local f_args		=args
	local f_args_num	=len f_args
	local line=""
	
	for i=0; i<f_args_num; ++i
		line=line+f_args[i]+" "
	end

	print(line)
end

_print("one", "two", "three")
